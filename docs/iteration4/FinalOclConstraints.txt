1. A task cannot have more than 20 subtasks
context Task
inv maxSubtasks:
  self.subtasks->size() <= 20

2. The number of open tasks without a due date should not exceed 50
context TaskRepository
inv maxUndatedOpenTasks:
  Task.allInstances()
    ->select(t | t.completionStatus = CompletionStatus::OPEN
             and t.dueDate.oclIsUndefined())
    ->size() <= 50

3. The limit for open tasks for each collaborator category must be a positive integer
context Collaborator
inv positiveTaskLimit:
  self.taskLimit > 0

4. No collaborator must be overloaded
context Collaborator
inv notOverloaded:
  Task.allInstances()
    ->select(t | t.completionStatus = CompletionStatus::OPEN
             and t.collaborators->includes(self))
    ->size() <= self.taskLimit
